#include <stdio.h>

int main() {
    int n, i;

    printf("Enter a number: ");
    scanf("%d", &n);

    if (n <= 1) {
        printf("No valid divisor\n");
        return 0;
    }

    for (i = 2; i <= n; i++) {
        if (n % i == 0) {
            printf("Smallest divisor = %d\n", i);
            break;
        }
    }

    return 0;
}
