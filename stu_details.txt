#include <stdio.h>

struct student {
    char name[20];
    int marks;
};

int main(){
    struct student s[3];
    int i;

    for(i=0;i<3;i++){
        printf("Enter name & marks: ");
        scanf("%s %d", s[i].name, &s[i].marks);
    }

    printf("\nResults:\n");
    for(i=0;i<3;i++){
        printf("%s - %d ", s[i].name, s[i].marks);
        if(s[i].marks>=40) printf("Pass\n");
        else printf("Fail\n");
    }

    return 0;
}
